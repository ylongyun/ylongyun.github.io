约瑟夫环问题

问题描述:编号为1.2..n的n个人按顺时针方向围坐一圈,每人持有一个密码(正整数)。
开始时任选一个整数作为报数上界m,从第一个人开始顺时针自1开始顺序报数，报到m时停止报数。
报m的人出列，将他的密码作为新的m值,从他在顺时针方向上的下一个人开始重新从1报数，如此下去,
直至所有的人全部出列为止。试设计一个程序,求出出列顺序。

思路

在问题描述可以知道，在每一次报数中我们需要找到第m个成员，并读取他的密码作为下一次报数的m值，由于m值可能大于n(所有人的总数)，因此在一次报数中会出现成员重复的现象，因此采用无头结点的循环单链表的数据结构来存放数据。由于成员出列后就会将成员所在的节点删除，所以解决问题的关键就是找到要出列节点的前一个节点，从而删除出列的节点，然后读取出列节点的密码作为下一次报数的m值。循环上述步骤直到只剩下一个节点，然后将此节点出列，问题解决。

代码实现

#include <stdio.h>
#include <stdlib.h>
typedef struct Node
{
    int order;//成员序号
    int code;//成员密码
    struct Node* next;//下一个结点的地址

}LinkList;

LinkList *create (int length)//尾插法创建循环链表
{
    int i;
    LinkList *head,*node,*pre;
    head=(LinkList*)malloc(sizeof(LinkList));
    head->next=NULL;
    pre=head;
    for (i=1;i<=length;i++)
    {
        node=(LinkList*)malloc(sizeof(LinkList));
        node->order=i;//结点空间开辟顺序即为此节点的序号
        scanf("%d",&node->code);//输入成员密码
        pre->next=node;
        pre=node;
    }
    node->next=head->next;//去除头结点
    return head->next;//返回第一个结点的地址
}

int yuesefu(LinkList* head,int m)
{
    int i;
    LinkList* pre,* node;
    pre = head;
    while(head->next != head)//循环结束条件：链表中只剩一个结点
    {
        if(m != 1)//此时
        {
            for(i = 1;i < (m-1);i++)
            {
            pre = pre->next; 
            }
            node = pre->next;

        }
        else{
            pre = head->next;
            while(pre->next != head){
                pre = pre->next;
            }
            node = head;
        }
        printf("%d\t",node->order);    //输出成员序号
        pre->next = node->next;
        head = node->next;
        pre = head;
        m = node->code;//读取密码
    }
    printf("%d\n",head->order);//输出最后一个出列的成员序号
    return 0;
}

int main()
{
    LinkList* head;
    head=create(7);
    yuesefu(head,20);
    return 0;
}
