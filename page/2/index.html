markdown语法

markdown基础语法及使用

标题

markdown支持两种形式的标题。
在标题前面加#，一级标题加1个#，二级标题加2个#，以此类推，最多加6个#，也就是6级标题。
例如：
# 一级标题
##二级标题
###三级标题
####四级标题

块引用

mrarkdown使用>字符作为块引用，且可以嵌套引用
例如：
>一级引用内容
>>二级引用内容

列表

markdown支持有序列表和无序列表。

  • 无序列表使用\ 、+、-。
    例如：
    `    title1-title+title`

显示效果

  • title1
  • title
  • title
  • 有序列表使用数字加句号。
    例如：
    1.title
    2.title
    3.title

显示效果

1.title
2.title
3.title

分割线

一行中只有三个或三个以上的 * 或 - 则会在这一行的位置上生成一条水平线。
例如：
***
-————–

##关键的空格
在配置站点配置文件时，Deployment下面每一个冒号后面都要有空格。
一定要有空格！一定要有空格！一定要有空格！（重要的事情说三遍！！！）

单链表

数据结构-单链表

在数据结构的学习中，第一个学的就是单链表，虽然在第一个学期的c语言中已经接触到这个东西，但实际上对链表还是一片模糊。在之前的几天里，慢慢地又把链表给学习了一遍，将单链表的基础操作自己写了一遍。如下。

1.定义结构体

在我理解中，链表就是由若干个结构体链接而成的，结构体内存放我们的数据，因此在使用链表之前要先根据你的需求定义结构体。 

typedef struct Node 
{
   int data;
   struct Node* next;

}LinkList;

2.初始化链表（尾插法建立链表）

在链表中添加头节点会简化对链表的操作，头节点内不储存任何数据，只起到一个确定链表地址的作用。 

LinkList* creatList(int n)  
{
   LinkList* head;  
   if (n == 0)//n为0的即生成一个空链表  
   {  
       head = (LinkList*)malloc(sizeof(LinkList));
       head->next = NULL;//封闭节点
       head->data = NULL;

   }
   else
   {
       int i;
       LinkList* node, * last;
       head = (LinkList*)malloc(sizeof(LinkList));
       last = head;
       for (i = 0;i < n;i++)   
       {
           node = (LinkList*)malloc(sizeof(LinkList));//为新节点开辟空间
           scanf("%d", &node->data);
           last->next = node;//将新节点连接到链表上
           last = node;//使新节点成为最后一个节点
           node->next = NULL;//封闭最后一个节点
       }

    }
    return(head);
}

3.输出链表

链表的输出需要注意节点的更替，且要完整的输出，不能出现遗漏。 

void print(LinkList* head)
{
    LinkList* node;
    if (head->next == NULL)//判断是否为空链表
        printf("there are nothing in the list.");
    else
   {
    node = head->next;//从第一个结点开始输出
    do {
        printf("%d\n", node->data);//输出节点内容
        node = node->next;//切换下一个节点
       } while (node->next != NULL);//输出最后一个节点后终止循环
   }  
   printf("%d\n", node->data);  
/*这里需要注意，当while循环结束时，最后一个节点地内容并没有输出，由while的结束循环标志可知此时node  
指向最后一个节点，因此我们只需要将node->data输出就可以了。*/
}

4.2按顺序查找

此操作用于确定链表中某一个元素的位置 

int find_order(LinkList* head, int num)  
{
    int i = 1;
    LinkList* node = head->next;
    while (node->data != num) //判断该节点数据是否为我们所查找的元素
    {
        i++;
        node = node->next;//切换节点
    }
    return(i);//返回所查找的元素的位置顺序
}

5.插入节点

更改链表的内容，在给定的序号后插入。 

void add(LinkList* head, int num)
{
    LinkList* pre,* node;
    int i=0;
    node = (LinkList*)malloc(sizeof(LinkList));//为新节点开辟空间
    pre = head;
    scanf("%d", &node->data);
    for (;i < num;i++) //确定插入位置前一个节点
    {
        pre = pre->next;
    }
    node->next = pre->next;//将插入位置之后的节点连接到新节点的后面
    pre->next = node;//将新节点连接到插入位置前一个节点的后面
}

6.删除节点

删除特定序号的节点，并释放该节点所占用的空间。 

void delete(LinkList* head, int num)
{
    LinkList* pre, * cur, * node;
    int i = 0;
    pre = head;
    for (;i < num-1;i++)
    {
        pre = pre->next;//确定被删除节点的前一个结点的位置
    }
    node = pre->next;//找到被删除节点的位置
    cur = node->next;//找到被删除节点的后一个节点的位置
    pre->next = cur;//将被删除的节点从链表中移除
    free(node);//释放被删除节点的内存

}

7.测量链表的长度

链表的长度不包括头节点，函数返回值为链表的长度。 

int size(LinkList* head)
{
    int i = 0;
    LinkList* node;
    node = head;
    if (node->next == NULL)//判断是否为空链表
    {
        return i;//是的话就直接返回0
    }
    else//不是的话就一个一个的数节点
    {
        while (1)
        {
            if (node->next != NULL) //
            {
                ++i;
                node = node->next;
            }
            else break;//没有节点就不数了
        }
        return (i);
    }
}
